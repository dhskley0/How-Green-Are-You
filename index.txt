  <script>
    const form = document.getElementById("quizForm");
    const resultDiv = document.getElementById("result");

    form.addEventListener("submit", async (e) => {
      e.preventDefault();

      const name = document.getElementById("name").value.trim();
      if (!name) return alert("Please enter your name!");

      let score = 0;
      for (let i = 1; i <= 5; i++) {
        const q = document.querySelector(`input[name="q${i}"]:checked`);
        if (q) score += parseInt(q.value);
      }

      const percentage = (score / 15) * 100;
      let category = "";

      if (percentage >= 71) category = "🌎 Planet Protector";
      else if (percentage >= 41) category = "🌿 Green Guardian";
      else category = "🌱 Budding Eco Explorer";

      resultDiv.innerHTML = `
        <h2 class="text-2xl font-bold mb-2">Hi, ${name}! 💚</h2>
        <p class="text-lg">You are a <strong>${category}</strong>!</p>
        <p>Your score: ${percentage.toFixed(0)}%</p>
        <button onclick="window.location.reload()" class="mt-4 bg-[#2f5d62] text-white px-4 py-2 rounded-lg">Try Again</button>
      `;

      form.style.display = "none";
      resultDiv.style.display = "block";

      // Send data to Google Sheet
      await fetch("https://script.google.com/macros/s/AKfycbyoR2LuSMewk8xcbcwu4UJ1xh-sGlZwRvqyWAcBTTnirnOJbbXsA0ZLNZKe691ooLCGiw/exec", {
        method: "POST",
        body: new FormData(form),
      });
    });
  </script>
